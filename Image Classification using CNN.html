# create the convolutional neural networks model 
cnn = models.Sequential([
    #cnn
    layers.Conv2D(filters = 32, kernel_size = (3, 3), activation = "relu", input_shape = (32, 32, 3)),
    layers.MaxPooling2D((2, 2)), # pooling (nb: you can either use the MaxPooling or the AvgPooling but the maxpooling is commonly used)
    
    layers.Conv2D(filters = 64, kernel_size = (3, 3), activation = "relu"),# no need of specifying the input_shape the second time
    layers.MaxPooling2D((2, 2)),
    
    # dense neural network
    layers.Flatten(),
    layers.Dense(64, activation = "relu"),
    layers.Dense(10, activation = "softmax") # the most commonly used activation is softmax
])
